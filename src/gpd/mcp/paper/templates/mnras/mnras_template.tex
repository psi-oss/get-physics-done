%# MNRAS template -- Monthly Notices of the Royal Astronomical Society
%# Jinja2 template with custom LaTeX-safe delimiters
%# The usenatbib class option turns on natbib inside mnras.cls so
%# \citet/\citep (from pandoc's --natbib output) resolve correctly.
\documentclass[usenatbib]{mnras}

\usepackage{graphicx}
\usepackage{amsmath}
\usepackage{amssymb}
\usepackage{wasysym}
\usepackage{hyperref}

%# Pandoc emits \tightlist for compact bullet lists; define as a no-op
%# fallback so lists don't crash the compile when the macro isn't provided
%# elsewhere.
\providecommand{\tightlist}{\setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}}

\title[\VAR{title}]{\VAR{title}}

\author[\BLOCK{for author in authors}\VAR{author.name}\BLOCK{if not loop.last}, \BLOCK{endif}\BLOCK{endfor}]{
\BLOCK{for author in authors}
\VAR{author.name}\BLOCK{if author.affiliation}$^{\VAR{loop.index}}$\BLOCK{endif}\BLOCK{if not loop.last}, \BLOCK{endif}
\BLOCK{endfor}
\BLOCK{if authors}\\
\BLOCK{endif}
\BLOCK{for author in authors}
\BLOCK{if author.affiliation}
$^{\VAR{loop.index}}$\VAR{author.affiliation}\\
\BLOCK{endif}
\BLOCK{endfor}
}

\begin{document}

\date{\today}

\maketitle

\begin{abstract}
\VAR{abstract}
\end{abstract}

\BLOCK{for section in sections}
\section{\VAR{section.title}}
\BLOCK{if section.label}\label{sec:\VAR{section.label}}\BLOCK{endif}
\VAR{section.content}
\BLOCK{endfor}

\BLOCK{for fig in figures}
\BLOCK{if fig.double_column}
\begin{figure*}
\BLOCK{else}
\begin{figure}
\BLOCK{endif}
\centering
\includegraphics[width=\VAR{fig.width}]{\VAR{fig.path}}
\caption{\VAR{fig.caption}}
\label{fig:\VAR{fig.label}}
\BLOCK{if fig.double_column}
\end{figure*}
\BLOCK{else}
\end{figure}
\BLOCK{endif}
\BLOCK{endfor}

\BLOCK{if acknowledgments}
\section*{Acknowledgements}
\VAR{acknowledgments}
\BLOCK{endif}

\BLOCK{if appendix_sections}
\appendix
\BLOCK{for section in appendix_sections}
\section{\VAR{section.title}}
\BLOCK{if section.label}\label{sec:\VAR{section.label}}\BLOCK{endif}
\VAR{section.content}
\BLOCK{endfor}
\BLOCK{endif}

\bibliographystyle{mnras}
\bibliography{\VAR{bib_file}}

\BLOCK{if attribution_footer}
\begin{center}
{\footnotesize\textit{\VAR{attribution_footer}}}
\end{center}
\BLOCK{endif}

\end{document}
