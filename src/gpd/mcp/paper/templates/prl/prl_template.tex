%# PRL template -- RevTeX 4.2
%# Jinja2 template with custom LaTeX-safe delimiters
%# Pass 'natbib' as a revtex4-2 class option (NOT \usepackage{natbib}).
%# revtex4-2 loads natbib internally when the class option is set; layering
%# \usepackage{natbib} on top causes option-clash warnings and can break on
%# some TeX Live versions. Without this option, revtex uses its own cite
%# engine and \citet/\citep (from pandoc's --natbib output) are undefined.
\documentclass[aps,prl,twocolumn,superscriptaddress,natbib]{revtex4-2}

\usepackage{graphicx}
\usepackage{amsmath}
\usepackage{amssymb}
\usepackage{wasysym}
\usepackage{hyperref}

%# Pandoc emits \tightlist for compact bullet lists; define as a no-op
%# fallback so lists don't crash the compile when the macro isn't provided
%# elsewhere.
\providecommand{\tightlist}{\setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}}

\begin{document}

\title{\VAR{title}}

\BLOCK{for author in authors}
\author{\VAR{author.name}}
\BLOCK{if author.email}\email{\VAR{author.email}}\BLOCK{endif}
\BLOCK{if author.affiliation}\affiliation{\VAR{author.affiliation}}\BLOCK{endif}
\BLOCK{endfor}

\date{\today}

\begin{abstract}
\VAR{abstract}
\end{abstract}

\maketitle

\BLOCK{for section in sections}
\section{\VAR{section.title}}
\BLOCK{if section.label}\label{sec:\VAR{section.label}}\BLOCK{endif}
\VAR{section.content}
\BLOCK{endfor}

\BLOCK{for fig in figures}
\BLOCK{if fig.double_column}
\begin{figure*}
\BLOCK{else}
\begin{figure}
\BLOCK{endif}
\centering
\includegraphics[width=\VAR{fig.width}]{\VAR{fig.path}}
\caption{\VAR{fig.caption}}
\label{fig:\VAR{fig.label}}
\BLOCK{if fig.double_column}
\end{figure*}
\BLOCK{else}
\end{figure}
\BLOCK{endif}
\BLOCK{endfor}

\BLOCK{if acknowledgments}
\begin{acknowledgments}
\VAR{acknowledgments}
\end{acknowledgments}
\BLOCK{endif}

\BLOCK{if appendix_sections}
\appendix
\BLOCK{for section in appendix_sections}
\section{\VAR{section.title}}
\BLOCK{if section.label}\label{sec:\VAR{section.label}}\BLOCK{endif}
\VAR{section.content}
\BLOCK{endfor}
\BLOCK{endif}

\bibliography{\VAR{bib_file}}

\BLOCK{if attribution_footer}
\begin{center}
{\footnotesize\textit{\VAR{attribution_footer}}}
\end{center}
\BLOCK{endif}

\end{document}
